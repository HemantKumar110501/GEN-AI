# GIT BRANCHING CLI COMMANDS

# Create a branch without switching
1. git branch branchname

# Create a branch and switch to that branch
1. git checkout -b branchname
2. git switch -c branchname

# only switch to already created branch
1. git switch branchname


# how to rename a branch
1. git branch -m branchname -> (respectedly)
2. git branch -M branchname -> (forcefully)

# merge a branch
1. git merge branchname

# delete a branch
1. git branch -d branchname > (respectedly)
2. git branch -D branchname -> (forcefully)
